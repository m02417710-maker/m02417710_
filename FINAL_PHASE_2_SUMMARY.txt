   $ cp client_src_index.js client/src/index.js
   $ cp client_src_App.js client/src/App.js
   $ cp client_src_redux_store.js client/src/redux/store.js
   $ cp client_redux_slices_stock.js client/src/redux/slices/stockSlice.js
   $ cp client_redux_slices_user.js client/src/redux/slices/userSlice.js

   $ cp client_pages_stocks.js client/src/pages/StocksPage.js


2️⃣ ثبت المكتبات:

   $ cd server && npm install
   $ cd ../client && npm install


3️⃣ شغّل التطبيق:

   # Terminal 1 - الخادم
   $ cd server && npm run dev
   
   # Terminal 2 - العميل
   $ cd client && npm start
   
   # Terminal 3 - Docker (اختياري)
   $ docker-compose up -d
